196MoFEMErrorCode VolumeElementForcesAndSourcesCore::calculateVolumeAndJacobian() {
197 MoFEMFunctionBegin;
198 const auto ent = numeredEntFiniteElementPtr->getEnt();
199 const auto type = numeredEntFiniteElementPtr->getEntType();
200 CHKERR mField.get_moab().get_connectivity(ent, conn, num_nodes, true);
201 coords.resize(3 * num_nodes, false);
202 CHKERR mField.get_moab().get_coords(conn, num_nodes, &*coords.data().begin());
203
204 auto get_tet_t_diff_n = [&]() {
205 return FTensor::Tensor1<FTensor::PackPtr<const double *, 3>, 3>(
206 &Tools::diffShapeFunMBTET[0], &Tools::diffShapeFunMBTET[1],
207 &Tools::diffShapeFunMBTET[2]);
208 };
209
210 auto get_hex_t_diff_n = [&]() {
211 return FTensor::Tensor1<FTensor::PackPtr<const double *, 3>, 3>(
212 &Tools::diffShapeFunMBHEXAtCenter[0],
213 &Tools::diffShapeFunMBHEXAtCenter[1],
214 &Tools::diffShapeFunMBHEXAtCenter[2]);
215 };
216
217 auto get_t_diff_n = [&]() {
218 if (type == MBTET)
219 return get_tet_t_diff_n();
220 return get_hex_t_diff_n();
221 };
222
223 auto t_diff_n = get_t_diff_n();
224
225 FTensor::Tensor1<FTensor::PackPtr<double *, 3>, 3> t_coords(
226 &coords[0], &coords[1], &coords[2]);
227 FTensor::Index<'i', 3> i;
228 FTensor::Index<'j', 3> j;
229 jAc.clear();
230
231 for (size_t n = 0; n != num_nodes; ++n) {
232 tJac(i, j) += t_coords(i) * t_diff_n(j);
233 ++t_coords;
234 ++t_diff_n;
235 }
236 CHKERR determinantTensor3by3(tJac, vOlume);
237 CHKERR invertTensor3by3(tJac, vOlume, tInvJac);
238
239 if (type == MBTET)
240 vOlume *= G_TET_W1[0] / 6.;
241
242 MoFEMFunctionReturn(0);
243}

361MoFEMErrorCode VolumeElementForcesAndSourcesCore::transformBaseFunctions() {
362 MoFEMFunctionBegin;
363
364 if (numeredEntFiniteElementPtr->getEntType() == MBTET) {
365 // OK that code is not nice.
366 MatrixDouble new_diff_n;
367 for (int b = AINSWORTH_LEGENDRE_BASE; b != LASTBASE; b++) {

370 EntitiesFieldData::EntData &data = dataH1.dataOnEntities[MBVERTEX][0];
371 if ((data.getDiffN(base).size1() == 4) &&
372 (data.getDiffN(base).size2() == 3)) {

374 const size_t nb_base_functions = 4;
375 new_diff_n.resize(nb_gauss_pts, 3 * nb_base_functions, false);
