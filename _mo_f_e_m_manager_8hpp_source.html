92MoFEMErrorCode MFManager::setMainField() {
93 MoFEMFunctionBegin;
94 // MoFEMFunctionReturnHot(0);
95
96 // Select base
97 enum bases { AINSWORTH, DEMKOWICZ, BERNSTEIN, LASBASETOPT };
98 const char *list_bases[] = {"ainsworth", "demkowicz", "bernstein"};
99 PetscInt choice_base_value = AINSWORTH;
100 CHKERR PetscOptionsGetEList(PETSC_NULL, NULL, "-base", list_bases,
101 LASBASETOPT, &choice_base_value, PETSC_NULL);
102
103 FieldApproximationBase base;
104 switch (choice_base_value) {
105 case AINSWORTH:
106 base = AINSWORTH_LEGENDRE_BASE;
107 MOFEM_LOG("WORLD", Sev::inform)
108 << "Set AINSWORTH_LEGENDRE_BASE for displacements";
109 break;
110 case DEMKOWICZ:
111 base = DEMKOWICZ_JACOBI_BASE;
112 MOFEM_LOG("WORLD", Sev::inform)
113 << "Set DEMKOWICZ_JACOBI_BASE for displacements";
114 break;
115 case BERNSTEIN:
116 base = AINSWORTH_BERNSTEIN_BEZIER_BASE;
117 MOFEM_LOG("WORLD", Sev::inform)
118 << "Set AINSWORTH_BERNSTEIN_BEZIER_BASE for displacements";
119 break;
120 default:
121 base = LASTBASE;
122 break;
123 }
124
125 // Add displacement field
126 CHKERR mField.add_field("U", H1, base, 3);
127
128 // Add field representing ho-geometry
129 CHKERR mField.add_field("MESH_NODE_POSITIONS", H1, AINSWORTH_LEGENDRE_BASE,
130 3);
131
132 // Add entities to field
133 CHKERR mField.add_ents_to_field_by_type(0, MBTET, "U");
134 CHKERR mField.add_ents_to_field_by_type(0, MBTET, "MESH_NODE_POSITIONS");
135
136 // Set approximation order to entities
137 if (base == AINSWORTH_BERNSTEIN_BEZIER_BASE)
138 CHKERR mField.set_field_order(0, MBVERTEX, "U", order);
139 else
140 CHKERR mField.set_field_order(0, MBVERTEX, "U", 1);
141 CHKERR mField.set_field_order(0, MBEDGE, "U", order);
142 CHKERR mField.set_field_order(0, MBTRI, "U", order);
143 CHKERR mField.set_field_order(0, MBTET, "U", order);
144
145 CHKERR mField.set_field_order(0, MBVERTEX, "MESH_NODE_POSITIONS", 1);
146 // CHKERR mField.set_field_order(0, MBEDGE, "MESH_NODE_POSITIONS", 2);
147 CHKERR mField.set_field_order(0, MBEDGE, "MESH_NODE_POSITIONS", 1);
148 CHKERR mField.set_field_order(0, MBTRI, "MESH_NODE_POSITIONS", 1);
149 CHKERR mField.set_field_order(0, MBTET, "MESH_NODE_POSITIONS", 1);
150
151 MoFEMFunctionReturn(0);
152}

154MoFEMErrorCode MFManager::defineDM() {
155 MoFEMFunctionBegin;
156
157 auto simple = mField.getInterface<Simple>();
158 DMType dm_name = "DMMOFEM";
159 // Register DM problem
160 CHKERR DMRegister_MoFEM(dm_name);
161 dM = createSmartDM(mField.get_comm(), dm_name);
162 CHKERR DMSetType(dM, dm_name);

164 // Create DM instance
165 CHKERR DMMoFEMCreateMoFEM(dM, &mField, simple->getProblemName().c_str(),
166 simple->getBitRefLevel());
167 CHKERR DMSetFromOptions(dM);
168
169 // mField.getInterface<ProblemsManager>()->buildProblemFromFields = PETSC_TRUE;
170
171 MoFEMFunctionReturn(0);

175MoFEMErrorCode MFManager::tsSetup() {
176 MoFEMFunctionBegin;
177
178 Simple *simple = mField.getInterface<Simple>();
179 PipelineManager *pipeline_mng = mField.getInterface<PipelineManager>();
180 ISManager *is_manager = mField.getInterface<ISManager>();
181
182 auto set_time_monitor = [&](auto solver) {
183 MoFEMFunctionBegin;
184 boost::shared_ptr<ForcesAndSourcesCore> null;
185 CHKERR DMMoFEMTSSetMonitor(dM, solver, simple->getDomainFEName(),
186 monitorPtr, null, null);
187 MoFEMFunctionReturn(0);
188 };
189
190 auto set_section_monitor = [&](auto solver) {
191 MoFEMFunctionBegin;
192 SNES snes;
193 CHKERR TSGetSNES(solver, &snes);
194 PetscViewerAndFormat *vf;
195 CHKERR PetscViewerAndFormatCreate(PETSC_VIEWER_STDOUT_WORLD,
196 PETSC_VIEWER_DEFAULT, &vf);
197 CHKERR SNESMonitorSet(
198 snes,
199 (MoFEMErrorCode(*)(SNES, PetscInt, PetscReal, void *))SNESMonitorFields,
200 vf, (MoFEMErrorCode(*)(void **))PetscViewerAndFormatDestroy);
201 MoFEMFunctionReturn(0);
202 };
203
204
205 auto set_dm_section = [&](auto dm) {
206 MoFEMFunctionBeginHot;
207 auto section = mField.getInterface<ISManager>()->sectionCreate(
208 simple->getProblemName());
209 CHKERR DMSetSection(dm, section);
210 MoFEMFunctionReturnHot(0);
211 };
212
213 auto scatter_create = [&](auto coeff, auto D) {
214 SmartPetscObj<IS> is;
215 CHKERR is_manager->isCreateProblemFieldAndRank(simple->getProblemName(),
216 ROW, "U", coeff, coeff, is);
217 int loc_size;
218 CHKERR ISGetLocalSize(is, &loc_size);
219 Vec v;
220 CHKERR VecCreateMPI(mField.get_comm(), loc_size, PETSC_DETERMINE, &v);
221 VecScatter scatter;
222 CHKERR VecScatterCreate(D, is, v, PETSC_NULL, &scatter);
223 return std::make_tuple(SmartPetscObj<Vec>(v),
224 SmartPetscObj<VecScatter>(scatter));
225 };
226
227 CHKERR set_dm_section(dM);
228
229 auto D = smartCreateDMVector(dM);
230 tSolver = MoFEM::createTS(mField.get_comm());
231
232 //FIXME: for explicit you need to implement proper mass matrix for rhs operators
233 if (isQuasiStatic || isExplicit) {
234 CHKERR TSSetSolution(tSolver, D);
235 } else {
236 CHKERR TSSetType(tSolver, TSALPHA2);
237 auto DD = smartVectorDuplicate(D);
238 CHKERR TS2SetSolution(tSolver, D, DD);
239 }
240 CHKERR set_section_monitor(tSolver);
241 CHKERR TSSetDM(tSolver, dM);
242
243 //default max time is 1
244 CHKERR TSSetMaxTime(tSolver, 1.0);
245 CHKERR TSSetExactFinalTime(tSolver, TS_EXACTFINALTIME_MATCHSTEP);
246 CHKERR TSSetFromOptions(tSolver);
247
248 for (int i = 0; i != 3; ++i)
249 vScatter[i] = scatter_create(i, D);
250 CHKERR set_time_monitor(tSolver);
251
252 MoFEMFunctionReturn(0);
253}

268MoFEMErrorCode MFManager::printMaxMin(
269 std::tuple<SmartPetscObj<Vec>, SmartPetscObj<VecScatter>> &tuple,
270 const std::string msg, Vec &ts_u, double time) {
271 MoFEMFunctionBegin;
272 double min_disp, max_disp;
273
274 CHKERR VecScatterBegin(std::get<1>(tuple), ts_u, std::get<0>(tuple),
275 INSERT_VALUES, SCATTER_FORWARD);
276 CHKERR VecScatterEnd(std::get<1>(tuple), ts_u, std::get<0>(tuple),
277 INSERT_VALUES, SCATTER_FORWARD);
278 CHKERR VecMax(std::get<0>(tuple), PETSC_NULL, &max_disp);
279 CHKERR VecMin(std::get<0>(tuple), PETSC_NULL, &min_disp);
280 MOFEM_LOG_C("WORLD", Sev::inform, "%s time %3.4e min %3.4e max %3.4e\n",
281 msg.c_str(), time, min_disp, max_disp);
282
283 if (atomTest == 3 && time == 1.00)
284 if (msg == "Ux") {
285 MOFEM_LOG("WORLD", Sev::inform)
286 << "Atom test min displacement X:" << min_disp;
287 if (abs(min_disp + 0.2220) > 1e-4)
288 SETERRQ(PETSC_COMM_SELF, MOFEM_ATOM_TEST_INVALID,
289 "atom test diverged!");
290 }
291
292 MoFEMFunctionReturn(0);
