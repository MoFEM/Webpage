240MoFEMErrorCode HMHNeohookean::OpSpatialPhysical::integrate(EntData &data) {
241 MoFEMFunctionBegin;
242
243 // std::array<double, 3> center;
244 // auto coords = getCoords();
245 // tetcircumcenter_tp(&coords[0], &coords[3], &coords[6], &coords[9],
246 // center.data(), nullptr, nullptr, nullptr);
247 // for (int d = 0; d!=3;++d)
248 // center[d] -= coords[d];
249 // double r = 0;
250 // for (int d = 0; d != 3; ++d)
251 // r += center[d] * center[d];
252 // r = std::sqrt(r);
253
254 auto neohookean_ptr =
255 boost::dynamic_pointer_cast<HMHNeohookean>(dataAtPts->physicsPtr);
256 if (!neohookean_ptr) {
257 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
258 "Pointer to HMHNeohookean is null");
259 }
260 auto [def_c10, def_K] =
261 neohookean_ptr->getMaterialParameters(getFEEntityHandle());
262
263 double c10 = def_c10 / neohookean_ptr->eqScaling;
264 double alpha_u = alphaU / neohookean_ptr->eqScaling;
265 double K = def_K / neohookean_ptr->eqScaling;
266
267 auto time_step = getTStimeStep();
268 double alpha_grad_u =
269 neohookean_ptr->alphaGradU / neohookean_ptr->eqScaling / time_step;
270
271 FTensor::Index<'L', size_symm> L;
272 auto t_L = symm_L_tensor();
273
274 constexpr auto t_kd_sym = FTensor::Kronecker_Delta_symmetric<int>();
275 constexpr auto t_kd = FTensor::Kronecker_Delta<int>();
276
277 int nb_dofs = data.getIndices().size();
278 int nb_integration_pts = data.getN().size1();
279 auto v = getVolume();
280 auto t_w = getFTensor0IntegrationWeight();
281 auto t_approx_P_adjoint_log_du =
282 getFTensor1FromMat<size_symm>(dataAtPts->adjointPdUAtPts);
283 auto t_u = getFTensor2SymmetricFromMat<3>(dataAtPts->stretchTensorAtPts);
284 auto t_grad_h1 = getFTensor2FromMat<3, 3>(dataAtPts->wGradH1AtPts);
285 auto t_dot_log_u =
286 getFTensor2SymmetricFromMat<3>(dataAtPts->logStretchDotTensorAtPts);
287 auto t_diff_u =
288 getFTensor4DdgFromMat<3, 3, 1>(dataAtPts->diffStretchTensorAtPts);
289 auto t_log_u =
290 getFTensor2SymmetricFromMat<3>(dataAtPts->logStretchTensorAtPts);
291 auto t_grad_log_u =
292 getFTensor2FromMat<6, 3>(dataAtPts->gradLogStretchDotTensorAtPts);
293 auto t_log_u2_h1 =
294 getFTensor2SymmetricFromMat<3>(dataAtPts->logStretch2H1AtPts);
295
296 auto t_eigen_vals = getFTensor1FromMat<3>(dataAtPts->eigenVals);
297 auto t_eigen_vecs = getFTensor2FromMat<3, 3>(dataAtPts->eigenVecs);
298 auto &nbUniq = dataAtPts->nbUniq;
299 auto t_nb_uniq =
300 FTensor::Tensor0<FTensor::PackPtr<int *, 1>>(nbUniq.data().data());
301
302 auto t_diff = diff_tensor();
303
304 FTENSOR_INDEX(SPACE_DIM, i);
305 FTENSOR_INDEX(SPACE_DIM, j);
306 FTENSOR_INDEX(SPACE_DIM, k);
307 FTENSOR_INDEX(SPACE_DIM, l);
308 FTENSOR_INDEX(SPACE_DIM, m);
309 FTENSOR_INDEX(SPACE_DIM, n);
310
311 auto get_ftensor2 = [](auto &v) {
312 return FTensor::Tensor1<FTensor::PackPtr<double *, size_symm>, size_symm>(
313 &v[0], &v[1], &v[2], &v[3], &v[4], &v[5]);
314 };
315
316 int nb_base_functions = data.getN().size2();
317 auto t_row_base_fun = data.getFTensor0N();
318 auto t_grad_base_fun = data.getFTensor1DiffN<3>();
319
320 auto no_h1 = [&]() {
321 MoFEMFunctionBegin;
322
323 for (int gg = 0; gg != nb_integration_pts; ++gg) {
324 double a = v * t_w;
325 ++t_w;
326
327 #ifndef NEOHOOKEAN_OFF_INVERSE_NO_H1
328 auto inv_f = [](auto v) { return EshelbianCore::f(-v); };
329 auto t_inv_u = EigenMatrix::getMat(t_eigen_vals, t_eigen_vecs, inv_f);
330 #endif // NEOHOOKEAN_OFF_INVERSE_NO_H1
331 ++t_eigen_vals;
332 ++t_eigen_vecs;
333 ++t_nb_uniq;
334
335 FTensor::Dg<double, SPACE_DIM, size_symm> t_Ldiff_u;
336 t_Ldiff_u(i, j, L) = t_diff_u(i, j, k, l) * t_L(k, l, L);
337 ++t_diff_u;
338
339 FTensor::Tensor2_symmetric<double, SPACE_DIM> t_Sigma_u;
340
341#ifndef NEOHOOKEAN_OFF_INVERSE_NO_H1
342 t_Sigma_u(i, j) = 2.0 * c10 * (t_u(i, j) - t_inv_u(i, j));
343#else
344 t_Sigma_u(i, j) = 2.0 * c10 * t_u(i, j);
345#endif // NEOHOOKEAN_OFF_INVERSE_NO_H1
346
347 const double tr = t_log_u(i, i);
348#ifndef NEOHOOKEAN_OFF_INVERSE_NO_H1
349 const double Sigma_J = K * tr;
350#else
351 const double Sigma_J = K * (tr - 2.0 * c10 / K);
352#endif // NEOHOOKEAN_OFF_INVERSE_NO_H1
353
354 FTensor::Tensor2_symmetric<double, SPACE_DIM> t_viscous_P;
355 t_viscous_P(i, j) = alpha_u * t_dot_log_u(i, j);
356 // That is only sheer part of the viscous stress
357 // alpha_u * (t_dot_log_u(i, j) -
358 // t_dot_log_u(k, l) * t_kd_sym(k, l) * t_kd_sym(i, j));
359
360 FTensor::Tensor1<double, size_symm> t_residual;
361 t_residual(L) = t_approx_P_adjoint_log_du(L);
362 t_residual(L) -= t_Ldiff_u(i, j, L) * t_Sigma_u(i, j);
363 t_residual(L) -= (t_L(i, j, L) * t_kd(i, j)) * Sigma_J;
364 t_residual(L) -= t_L(i, j, L) * t_viscous_P(i, j);
365 t_residual(L) *= a;
366
367 FTensor::Tensor2<double, size_symm, SPACE_DIM> t_grad_residual;
368 t_grad_residual(L, i) = alpha_grad_u * t_grad_log_u(L, i);
369 t_grad_residual(L, i) *= a;
370
371 ++t_approx_P_adjoint_log_du;
372 ++t_u;
373 ++t_log_u;
374 ++t_dot_log_u;
375 ++t_grad_log_u;
376
377 auto t_nf = getFTensor1FromPtr<size_symm>(&*nF.data().begin());
378 int bb = 0;
379 for (; bb != nb_dofs / size_symm; ++bb) {
380 t_nf(L) -= t_row_base_fun * t_residual(L);
381 t_nf(L) += t_grad_base_fun(i) * t_grad_residual(L, i);
382 ++t_nf;
383 ++t_row_base_fun;
384 ++t_grad_base_fun;
385 }
386 for (; bb != nb_base_functions; ++bb) {
387 ++t_row_base_fun;
388 ++t_grad_base_fun;
389 }
390 }
391
392 MoFEMFunctionReturn(0);
393 };
394
395 auto large = [&]() {
396 MoFEMFunctionBegin;
397
398 for (int gg = 0; gg != nb_integration_pts; ++gg) {
399 double a = v * t_w;
400 ++t_w;
401
402 FTensor::Tensor3<double, SPACE_DIM, SPACE_DIM, size_symm> t_Ldiff_u;
403 FTensor::Tensor2<double, SPACE_DIM, SPACE_DIM> t_h1;
404 t_h1(i, j) = t_grad_h1(i, j) + t_kd(i, j);
405 t_Ldiff_u(i, j, L) = (t_diff_u(i, m, k, l) * t_h1(m, j)) * t_L(k, l, L);
406 ++t_diff_u;
407 ++t_grad_h1;
408
409 FTensor::Tensor2<double, SPACE_DIM, SPACE_DIM> t_Sigma_u;
410 const double tr_h1 = t_log_u2_h1(i, j) * t_kd_sym(i, j) / 2;
411 const double tr_u = t_log_u(i, j) * t_kd_sym(i, j);
412 const double tr = tr_h1 + tr_u;
413
414 FTensor::Tensor2<double, SPACE_DIM, SPACE_DIM> t_total_u;
415 t_total_u(i, j) = t_u(i, m) * t_h1(m, j);
416#ifndef NEOHOOKEAN_OFF_INVERSE_LARGE
417 FTensor::Tensor2<double, SPACE_DIM, SPACE_DIM> t_inv_total_u;
418 invertTensor3by3(t_total_u, J, t_inv_total_u);
419#endif // NEOHOOKEAN_OFF_INVERSE_LARGE
420
421#ifndef NEOHOOKEAN_OFF_INVERSE_LARGE
422 t_Sigma_u(i, j) = 2.0 * c10 * (t_total_u(i, j) - t_inv_total_u(i, j));
423#else
424 t_Sigma_u(i, j) = 2.0 * c10 * t_total_u(i, j);
425#endif // NEOHOOKEAN_OFF_INVERSE_LARGE
426
427#ifndef NEOHOOKEAN_OFF_INVERSE_LARGE
428 const double Sigma_J = K * tr;
429#else
430 const double Sigma_J = K * (tr - 2.0 * c10 / K);
431#endif // NEOHOOKEAN_OFF_INVERSE_LARGE
432
433 FTensor::Tensor2_symmetric<double, SPACE_DIM> t_viscous_P;
434 t_viscous_P(i, j) = alpha_u * t_dot_log_u(i, j);
435 // That is only sheer part of the viscous stress
436 // (t_dot_log_u(i, j) -
437 // t_dot_log_u(k, l) * t_kd_sym(k, l) * t_kd_sym(i, j));
438
439 FTensor::Tensor1<double, size_symm> t_residual;
440 t_residual(L) = t_approx_P_adjoint_log_du(L);
441 t_residual(L) -= t_Ldiff_u(i, j, L) * t_Sigma_u(i, j);
442 t_residual(L) -= (t_L(i, j, L) * t_kd(i, j)) * Sigma_J;
443 t_residual(L) -= t_L(i, j, L) * t_viscous_P(i, j);
444 t_residual(L) *= a;
445
446 FTensor::Tensor2<double, size_symm, SPACE_DIM> t_grad_residual;
447 t_grad_residual(L, i) = alpha_grad_u * t_grad_log_u(L, i);
448 t_grad_residual(L, i) *= a;
449
450 ++t_approx_P_adjoint_log_du;
451 ++t_u;
452 ++t_log_u;
453 ++t_log_u2_h1;
454 ++t_dot_log_u;
455 ++t_grad_log_u;
456
457 auto t_nf = getFTensor1FromPtr<size_symm>(&*nF.data().begin());
458 int bb = 0;
459 for (; bb != nb_dofs / size_symm; ++bb) {
460 t_nf(L) -= t_row_base_fun * t_residual(L);
461 t_nf(L) += t_grad_base_fun(i) * t_grad_residual(L, i);
462 ++t_nf;
463 ++t_row_base_fun;
464 ++t_grad_base_fun;
465 }
466 for (; bb != nb_base_functions; ++bb) {
467 ++t_row_base_fun;
468 ++t_grad_base_fun;
469 }
470 }
471
472 MoFEMFunctionReturn(0);
473 };
474
475 switch (EshelbianCore::gradApproximator) {
476 case NO_H1_CONFIGURATION:
477 CHKERR no_h1();
478 break;
479 case LARGE_ROT:
480 case MODERATE_ROT:
481 CHKERR large();
482 break;
483 default:
484 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
485 "gradApproximator not handled");
486 };
487
488 MoFEMFunctionReturn(0);
489}
