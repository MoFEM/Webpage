417/**
418 * @brief Assemble the left hand side, i.e. tangent matrix
419 * @ingroup adoc_plasticity
420 *
421 * @tparam DIM dimension of the problem
422 * @tparam A assembly type
423 * @tparam I integration type
424 * @tparam DomainEleOp operator type
425 * @param m_field
426 * @param field_name
427 * @param pip
428 * @param block_name esh block name caring material parameters
429 * @param common_data_ptr
430 * @param cp_ptr
431 * @return MoFEMErrorCode
432 */
433template <int DIM, AssemblyType A, IntegrationType I, typename DomainEleOp>
434MoFEMErrorCode opFactoryDomainRhs(
435 MoFEM::Interface &m_field, string field_name, Pip &pip,
436 std::string block_name, boost::shared_ptr<CommonData> common_data_ptr,
437 boost::shared_ptr<ClosestPointProjection> cp_ptr, Sev sev = Sev::inform) {
438 MoFEMFunctionBegin;
439 using P = ADOLCPlasticityIntegrators<DomainEleOp>;
440 CHKERR cp_ptr->addMatBlockOps(m_field, pip, block_name, sev);
441 pip.push_back(
442 getRawPtrOpCalculateStress<DIM, SMALL_STRAIN>(m_field, common_data_ptr, cp_ptr, false));
443 pip.push_back(new typename P::template Assembly<A>::template OpRhs<DIM, I>(
444 field_name, common_data_ptr));
445 MoFEMFunctionReturn(0);
446}

579//! [OpRhsImpl integrate]
580template <int DIM, typename AssemblyDomainEleOp>
581MoFEMErrorCode OpRhsImpl<DIM, GAUSS, AssemblyDomainEleOp>::iNtegrate(
582 EntitiesFieldData::EntData &data) {
583 MoFEMFunctionBegin;
584 using OP = AssemblyDomainEleOp;
585 FTensor::Index<'i', 3> i;
586 FTensor::Index<'j', 3> j;
587 double *w_ptr = &(OP::getGaussPts()(DIM, 0));
588 // Calulate tensorial base functions. Apply bBar method when needed
589 auto t_diff = MakeB::getFTensor2SymmetricDiffBase(
590 data, baseStorage, commonDataPtr->bBar, OP::getGaussPts().size2(), w_ptr,
591 FTensor::Number<DIM>());
592
593 auto t_stress = getFTensor2SymmetricFromMat<3>(commonDataPtr->stressMatrix);
594
595 const auto vol = OP::getMeasure();
596 auto t_w = OP::getFTensor0IntegrationWeight();
597 for (int gg = 0; gg != OP::nbIntegrationPts; gg++) {
598 double alpha = vol * t_w;
599 for (int bb = 0; bb != OP::nbRows; ++bb) {
600 OP::locF[bb] += alpha * (t_stress(i, j) * t_diff(i, j));
601 ++t_diff;
602 }
603 ++t_w;
604 ++t_stress;
605 }
606 MoFEMFunctionReturn(0);
607}

619template <int DIM, typename AssemblyDomainEleOp>
620MoFEMErrorCode OpLhsImpl<DIM, GAUSS, AssemblyDomainEleOp>::iNtegrate(
621 DataForcesAndSourcesCore::EntData &row_data,
622 DataForcesAndSourcesCore::EntData &col_data) {
623 MoFEMFunctionBegin;
624 using OP = AssemblyDomainEleOp;
625
626 double *w_ptr = &(OP::getGaussPts()(DIM, 0));
627 auto t_diff_row = MakeB::getFTensor2SymmetricDiffBase(
628 row_data, baseRowStorage, commonDataPtr->bBar, OP::getGaussPts().size2(),
629 w_ptr, FTensor::Number<DIM>());
630 auto t_diff_col = MakeB::getFTensor2SymmetricDiffBase(
631 col_data, baseColStorage, commonDataPtr->bBar, OP::getGaussPts().size2(),
632 w_ptr, FTensor::Number<DIM>());
633
634 auto get_ftensor2_symmetric = [&](auto &storage, const auto gg,
635 const auto rr) {
636 return FTensor::Tensor2_symmetric<FTensor::PackPtr<double *, 6>, 3>{
637 &storage(gg, 6 * rr + 0), &storage(gg, 6 * rr + 1),
638 &storage(gg, 6 * rr + 2), &storage(gg, 6 * rr + 3),
639 &storage(gg, 6 * rr + 4), &storage(gg, 6 * rr + 5)};
640 };
641
642 FTensor::Index<'i', 3> i;
643 FTensor::Index<'j', 3> j;
644 FTensor::Index<'k', 3> k;
645 FTensor::Index<'l', 3> l;
646
647 auto t_Cep =
648 getFTensor4DdgFromMat<3, 3, 1>(commonDataPtr->materialTangentOperator);
649 const auto vol = OP::getMeasure();
650 auto t_w = OP::getFTensor0IntegrationWeight();
651 for (int gg = 0; gg != OP::nbIntegrationPts; ++gg) {
652 const double alpha = vol * t_w;
653 ++t_w;
654
