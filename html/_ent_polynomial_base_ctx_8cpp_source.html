38MoFEMErrorCode EntPolynomialBaseCtx::setBase() {
39 MoFEMFunctionBeginHot;
40 switch (bAse) {
41 case AINSWORTH_LEGENDRE_BASE:
42 switch (sPace) {
43 case NOSPACE:
44 case NOFIELD:
45 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY, "Makes no sense");
46 case H1:
47 case HCURL:
48 case HDIV:
49 case L2:
50 basePolynomialsType0 = Legendre_polynomials;
51 break;
52 default:
53 SETERRQ1(PETSC_COMM_SELF, MOFEM_NOT_IMPLEMENTED,
54 "Not implemented for this space", FieldSpaceNames[sPace]);
55 }
56 break;
57 case AINSWORTH_LOBATTO_BASE:
58 switch (sPace) {
59 case NOSPACE:
60 case NOFIELD:
61 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY, "Makes no sense");
62 case H1:
63 case HCURL:
64 case HDIV:
65 case L2:
66 basePolynomialsType0 = LobattoKernel_polynomials;
67 break;
68 default:
69 SETERRQ1(PETSC_COMM_SELF, MOFEM_NOT_IMPLEMENTED,
70 "Not implemented for this space", FieldSpaceNames[sPace]);
71 }
72 break;
73 case AINSWORTH_BERNSTEIN_BEZIER_BASE:
74 if(fieldName.empty())
75 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY, "Field name not set");
76 switch (sPace) {
77 case NOSPACE:
78 case NOFIELD:
79 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY, "Makes no sense");
80 case H1:
81 case L2:
82 break;
83 default:
84 SETERRQ1(PETSC_COMM_SELF, MOFEM_NOT_IMPLEMENTED,
85 "Not implemented for this space", FieldSpaceNames[sPace]);
86 }
87 break;
88 case DEMKOWICZ_JACOBI_BASE:
89 switch (sPace) {
90 case NOSPACE:
91 case NOFIELD:
92 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY, "Makes no sense");
93 case H1:
94 case HCURL:
95 case HDIV:
96 case L2:
97 basePolynomialsType0 = Legendre_polynomials;
98 basePolynomialsType1 = Jacobi_polynomials;
99 break;
100 default:
101 SETERRQ1(PETSC_COMM_SELF, MOFEM_NOT_IMPLEMENTED,
102 "Not implemented for this space",
103 FieldSpaceNames[sPace]);
104 }
105 break;
106 case USER_BASE:
107 break;
108 default:
109 SETERRQ1(PETSC_COMM_SELF, MOFEM_NOT_IMPLEMENTED,
110 "Not implemented for this base <%s>",
111 ApproximationBaseNames[bAse]);
112 }
113 MoFEMFunctionReturnHot(0);
114}
