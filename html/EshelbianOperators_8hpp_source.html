160 using OpAssembleVolume::OpAssembleVolume;
161 MoFEMErrorCode doWork(int side, EntityType type, EntData &data) {
162 MoFEMFunctionBegin;
163 SETERRQ(PETSC_COMM_SELF, MOFEM_NOT_IMPLEMENTED,

168 MoFEMErrorCode doWork(int row_side, int col_side, EntityType row_type,
169 EntityType col_type, EntData &row_data,
170 EntData &col_data) {
171 MoFEMFunctionBegin;
172 if (row_data.getIndices().empty())
173 MoFEMFunctionReturnHot(0);
174 if (col_data.getIndices().empty())
175 MoFEMFunctionReturnHot(0);
176 K.resize(row_data.getIndices().size(), col_data.getIndices().size(), false);
177 K.clear();
178 CHKERR integrate(row_data, col_data);
179
180 constexpr bool run_psd = true;
181
182 if (run_psd) {
183
184 auto symmetrize = [](MatrixDouble &mat) {
185 MatrixDouble r = (mat + trans(mat)) / 2.0;
186 return r;
187 };
188 auto check_positive_definite = [](MatrixDouble &mat) {
189 int n = mat.size1();
190 return (lapack_dpotrf('U', n, mat.data().data(), n) == 0);
191 };
192 auto frobenius_norm = [](MatrixDouble &mat) {
193 double sum = 0.0;
194 for (std::size_t i = 0; i < mat.size1(); ++i)
195 for (std::size_t j = 0; j < mat.size2(); ++j)
196 sum += mat(i, j) * mat(i, j);
197 return std::sqrt(sum);
198 };
199
200 auto higham_method_one_pass = [&](MatrixDouble &mat) {
201 int n = mat.size1();
202 const int lda = n;
203 const int size = (n + 2) * n;
204 int lwork = size;
205 double *work = new double[size];
206
207 VectorDouble eig(n, 0.0);
208 MatrixDouble eigen_vec = symmetrize(mat);
209 if (lapack_dsyev('V', 'U', n, eigen_vec.data().data(), lda,
210 eig.data().data(), work, lwork) > 0)
211 CHK_THROW_MESSAGE(MOFEM_INVALID_DATA,
212 "The algorithm failed to compute eigenvalues.");
213 delete[] work;
214
215 double max_e = std::abs(eig[0]);
216 for (auto e : eig) {
217 max_e = std::max(std::abs(e), max_e);
218 }
219 constexpr auto eps = std::numeric_limits<float>::epsilon();
220 auto min_e = max_e * eps;
221 for (auto &e : eig) {
222 e = std::max(e, min_e);
223 }
224
225 auto diagonal_matrix =
226 ublas::diagonal_matrix<double, ublas::row_major,
227 VecAllocator<double>>(n, eig.data());
228 MatrixDouble symm_mat;
229 symm_mat = prod(trans(eigen_vec), diagonal_matrix);
230 symm_mat = prod(symm_mat, eigen_vec);
231
232 MatrixDouble hat_m = (mat + symm_mat) / 2.0;
233
234 return hat_m;
235 };
236
237 auto hat_k = higham_method_one_pass(K);
238 K.swap(hat_k);
239 }
240
241 CHKERR assemble(row_side, col_side, row_type, col_type, row_data, col_data);
242 MoFEMFunctionReturn(0);
243 }
244
245private:
246};
