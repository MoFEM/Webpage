52  for (unsigned int gg = 0; gg != nb_integration_pts; ++gg) {
53 
54  FTensor::Tensor2<double, 3, 3> t_h1;
55

64  physicsPtr->get_h()(i, j) = iu(i, k) * t_h1(k, j);
65  break;
66  case SMALL_ROT:
67  physicsPtr->get_h()(i, j) = iu(i, j);
68  break;
69  };
70

72  physicsPtr->activeVariables.size(),
73  &physicsPtr->activeVariables[0],
74  &physicsPtr->dependentVariablesPiola[0]);
75  if (r < 0) { // function is locally analytic
76  SETERRQ1(PETSC_COMM_SELF, MOFEM_OPERATION_UNSUCCESSFUL,
77  "ADOL-C function evaluation with error r = %d", r);
78  }

81  case NO_H1_CONFIGURATION:
82  r_P(i, j) = physicsPtr->get_P()(i, j);
83  break;
84  case LARGE_ROT:
85  case MODERATE_ROT: {

88  r_P(k, l) = physicsPtr->get_P()(i, j) * t_h1_du(i, j, k, l);
89  }; break;
90  case SMALL_ROT:

93  };

103  MoFEMFunctionBegin;
104  FTensor::Index<'i', 3> i;
105  FTensor::Index<'j', 3> j;

114  const int number_of_dependent_variables =
115  physicsPtr->dependentVariablesPiola.size();
116  std::vector<double *> jac_ptr(number_of_dependent_variables);
117  for (unsigned int n = 0; n != number_of_dependent_variables; ++n) {
118  jac_ptr[n] =

120  ->dependentVariablesPiolaDirevatives[n *

122  }
123 
124  const auto nb_integration_pts = getGaussPts().size2();
125 
126  auto create_data_mat = [nb_integration_pts](auto &m) {
127  m.resize(9, nb_integration_pts, false);
128  };
129 
130  dataAtPts->P_du.resize(81, nb_integration_pts, false);
131 
132  auto iu = getFTensor2SymmetricFromMat<3>(dataAtPts->stretchTensorAtPts);
133  auto t_grad_h1 = getFTensor2FromMat<3, 3>(dataAtPts->wGradH1AtPts);
134  auto r_P_du = getFTensor4FromMat<3, 3, 3, 3>(dataAtPts->P_du);
135 
136  FTensor::Number<0> N0;
137  FTensor::Number<1> N1;
138  FTensor::Number<2> N2;
139 
140  constexpr auto t_kd = FTensor::Kronecker_Delta<int>();
141 
142  for (unsigned int gg = 0; gg != nb_integration_pts; ++gg) {
143 
144  FTensor::Tensor2<double, 3, 3> t_h1;
145 
146  switch (EshelbianCore::gradApperoximator) {
147  case NO_H1_CONFIGURATION:
148  t_h1(i, j) = t_kd(i, j);
149  case LARGE_ROT:
150  case MODERATE_ROT:
151  t_h1(i, j) = t_grad_h1(i, j) + t_kd(i, j);
152  physicsPtr->get_h()(i, j) = iu(i, k) * t_h1(k, j);
153  break;
154  case SMALL_ROT:
155  physicsPtr->get_h()(i, j) = iu(i, j);
156  break;
157  };
158 
159  // play recorder for jacobians
160  int r = ::jacobian(tAg, number_of_dependent_variables,
161  number_of_active_variables,
162  &physicsPtr->activeVariables[0], &jac_ptr[0]);
163  if (r < 0) {
164  SETERRQ(PETSC_COMM_SELF, MOFEM_OPERATION_UNSUCCESSFUL,
165  "ADOL-C function evaluation with error");
166  }
167 
168  FTensor::Tensor4<double, 3, 3, 3, 3> t_P_dh_tmp;
169  t_P_dh_tmp(i, j, N0, k) = physicsPtr->get_P_dh0()(i, j, k);
170  t_P_dh_tmp(i, j, N1, k) = physicsPtr->get_P_dh1()(i, j, k);
171  t_P_dh_tmp(i, j, N2, k) = physicsPtr->get_P_dh2()(i, j, k);
172 
173  switch (EshelbianCore::gradApperoximator) {
174  case NO_H1_CONFIGURATION:
175  case LARGE_ROT:
176  case MODERATE_ROT: {
177  FTensor::Tensor4<double, 3, 3, 3, 3> t_h1_du;
178  t_h1_du(i, j, m, n) = t_kd(i, m) * (t_kd(k, n) * t_h1(k, j));
179  r_P_du(k, l, m, n) =
180  (t_P_dh_tmp(i, j, o, p) * t_h1_du(o, p, m, n)) * t_h1_du(i, j, k, l);
181  } break;
182  case SMALL_ROT:
183  r_P_du(i, j, m, n) = t_P_dh_tmp(i, j, m, n);
184  break;
185  };
186 
187  ++iu;
188  ++t_grad_h1;
189  ++r_P_du;
190  }
191  MoFEMFunctionReturn(0);

260  MoFEMErrorCode recordTape(const int tape, DTensor2Ptr *t_h_ptr) {
261  MoFEMFunctionBegin;
262  FTensor::Index<'i', 3> i;
263  FTensor::Index<'j', 3> j;
264  FTensor::Index<'I', 3> I;
265  FTensor::Index<'J', 3> J;
266  FTensor::Number<0> N0;
267  FTensor::Number<1> N1;
268  FTensor::Number<2> N2;
269 
270  CHKERR getOptions();
271 
272  auto t_kd = FTensor::Kronecker_Delta_symmetric<int>();
273 
274  auto ih = get_h();
275  // auto iH = get_H();
276  if (t_h_ptr)
277  ih(i, j) = (*t_h_ptr)(i, j);
278  else {
279  ih(i, j) = t_kd(i, j);
280  }
281 
282  auto r_P = get_P();
283 
284  enableMinMaxUsingAbs();
285  trace_on(tape);
286 
287  // Set active variables to ADOL-C
288  th(i, j) <<= get_h()(i, j);
289 
290  tF(i, I) = th(i, I);
291  CHKERR determinantTensor3by3(tF, detF);
292  CHKERR invertTensor3by3(tF, detF, tInvF);
293 
294  // Deformation and strain
295  tC(I, J) = tF(i, I) * tF(i, J);
296  tE(I, J) = tC(I, J);
297  tE(N0, N0) -= 1;
298  tE(N1, N1) -= 1;
299  tE(N2, N2) -= 1;
300  tE(I, J) *= 0.5;
301 
302  // Energy
303  trE = tE(I, I);
304  energy = 0.5 * lambda * trE * trE + mu * (tE(I, J) * tE(I, J));
305 
306  // Stress Piola II
307  tS(I, J) = tE(I, J);
308  tS(I, J) *= 2 * mu;
309  tS(N0, N0) += lambda * trE;
310  tS(N1, N1) += lambda * trE;
311  tS(N2, N2) += lambda * trE;
312  // Stress Piola I
313  tP(i, J) = tF(i, I) * tS(I, J);
314 
315  // Set dependent variables to ADOL-C
316  tP(i, j) >>= r_P(i, j);
317 
318  trace_off();
319 
320  MoFEMFunctionReturn(0);
321  }

603  boost::shared_ptr<MatrixDouble> mat_axiator_D_ptr,
604  boost::shared_ptr<MatrixDouble> mat_deviator_D_ptr,
605  double bulk_modulus_K, double shear_modulus_G) {
606  MoFEMFunctionBegin;
607  //! [Calculate elasticity tensor]
608  auto set_material_stiffness = [&]() {
609  FTensor::Index<'i', SPACE_DIM> i;
610  FTensor::Index<'j', SPACE_DIM> j;
611  FTensor::Index<'k', SPACE_DIM> k;
612  FTensor::Index<'l', SPACE_DIM> l;
613  constexpr auto t_kd = FTensor::Kronecker_Delta_symmetric<int>();
614  double A = (SPACE_DIM == 2)
615  ? 2 * shear_modulus_G /
616  (bulk_modulus_K + (4. / 3.) * shear_modulus_G)
617  : 1;
618  auto t_D = getFTensor4DdgFromMat<SPACE_DIM, SPACE_DIM, 0>(*mat_D_ptr);
619  auto t_axiator_D =
620  getFTensor4DdgFromMat<SPACE_DIM, SPACE_DIM, 0>(*mat_axiator_D_ptr);
621  auto t_deviator_D =
622  getFTensor4DdgFromMat<SPACE_DIM, SPACE_DIM, 0>(*mat_deviator_D_ptr);
623  t_axiator_D(i, j, k, l) = A *
624  (bulk_modulus_K - (2. / 3.) * shear_modulus_G) *
625  t_kd(i, j) * t_kd(k, l);
626  t_deviator_D(i, j, k, l) =
627  2 * shear_modulus_G * ((t_kd(i, k) ^ t_kd(j, l)) / 4.);
628  t_D(i, j, k, l) = t_axiator_D(i, j, k, l) + t_deviator_D(i, j, k, l);
629  };
630  //! [Calculate elasticity tensor]
631  constexpr auto size_symm = (SPACE_DIM * (SPACE_DIM + 1)) / 2;
632  mat_D_ptr->resize(size_symm * size_symm, 1);
633  mat_axiator_D_ptr->resize(size_symm * size_symm, 1);
634  mat_deviator_D_ptr->resize(size_symm * size_symm, 1);
635  set_material_stiffness();
636  MoFEMFunctionReturn(0);
637  }
