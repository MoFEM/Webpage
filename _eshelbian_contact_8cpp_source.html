30OpConstrainBoundaryRhs::iNtegrate(EntitiesFieldData::EntData &data) {
31 MoFEMFunctionBegin;
32
33 FTensor::Index<'i', 3> i;
34 FTensor::Index<'j', 3> j;
35 FTensor::Index<'k', 3> k;
36 FTensor::Index<'l', 3> l;
37
38 const size_t nb_gauss_pts = getGaussPts().size2();
39
40 // #ifndef NDEBUGbase
41 if (commonDataPtr->contactDisp.size2() != nb_gauss_pts) {
42 SETERRQ2(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
43 "Wrong number of integration pts %d != %d",
44 commonDataPtr->contactDisp.size2(), nb_gauss_pts);
45 }
46 // #endif // !NDEBUG
47
48 auto &nf = locF;
49 locF.clear();
50
51 auto t_w = getFTensor0IntegrationWeight();
52 auto t_disp = getFTensor1FromMat<3>(commonDataPtr->contactDisp);
53 auto t_traction = getFTensor1FromMat<3>(commonDataPtr->contactTraction);
54 auto t_coords = getFTensor1CoordsAtGaussPts();
55 auto t_normal_at_pts = getFTensor1NormalsAtGaussPts();
56
57 auto next = [&]() {
58 ++t_w;
59 ++t_disp;
60 ++t_traction;
61 ++t_coords;
62 ++t_normal_at_pts;
63 };
64
65 size_t nb_base_functions = data.getN().size2() / 3;
66 auto t_base = data.getFTensor1N<3>();
67 for (size_t gg = 0; gg != nb_gauss_pts; ++gg) {
68
69 FTensor::Tensor1<double, 3> t_normal;
70 t_normal(i) = t_normal_at_pts(i) / t_normal_at_pts.l2();
71
72 auto t_nf = getFTensor1FromPtr<3>(&nf[0]);
73 const double alpha = t_w * getMeasure();
74
75 FTensor::Tensor1<double, 3> t_spatial_coords{0., 0., 0.};
76 t_spatial_coords(i) = t_coords(i) + t_disp(i);
77
78 auto ts_t = getTStime();
79
80 auto sdf =
81 surfaceDistanceFunction(ts_t, t_spatial_coords(0), t_spatial_coords(1),
82 t_spatial_coords(2), 0, 0, 0);
83
84 auto t_grad_sdf = gradSurfaceDistanceFunction(ts_t, t_spatial_coords(0),
85 t_spatial_coords(1),
86 t_spatial_coords(2), 0, 0, 0);
87
88 auto tn = t_traction(i) * t_grad_sdf(i);
89 auto c = ContactOps::constrain(sdf, tn);
90
91 FTensor::Tensor2<double, 3, 3> t_cP;
92 t_cP(i, j) = (c * t_grad_sdf(i)) * t_grad_sdf(j);
93 FTensor::Tensor2<double, 3, 3> t_cQ;
94 t_cQ(i, j) = kronecker_delta(i, j) - t_cP(i, j);
95
96 FTensor::Tensor1<double, 3> t_disp_e;
97 t_disp_e(i) = t_disp(i) - ContactOps::cn_contact * t_traction(i);
98
99 FTensor::Tensor1<double, 3> t_rhs;
100 t_rhs(i) = -c * sdf * t_grad_sdf(i)
101
102 +
103
104 t_cP(i, j) * t_disp(j)
105
106 +
107
108 t_cQ(i, j) * t_disp_e(j);
109
110 size_t bb = 0;
111 for (; bb != nbRows / 3; ++bb) {
112 const double row_base = t_base(i) * t_normal(i);
113 const double beta = alpha * row_base;
114 t_nf(i) -= beta * t_rhs(i);
115 ++t_nf;
116 ++t_base;
117 }
118 for (; bb < nb_base_functions; ++bb)
119 ++t_base;
120
121 next();
122 }
123
124 MoFEMFunctionReturn(0);
125}
