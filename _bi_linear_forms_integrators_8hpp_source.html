985template <int SPACE_DIM, int S, typename OpBase>
986MoFEMErrorCode
987OpGradSymTensorGradImpl<1, SPACE_DIM, SPACE_DIM, S, GAUSS, OpBase>::iNtegrate(
988 EntitiesFieldData::EntData &row_data,
989 EntitiesFieldData::EntData &col_data) {
990 MoFEMFunctionBegin;
991
992 const size_t nb_row_dofs = row_data.getIndices().size();
993 const size_t nb_col_dofs = col_data.getIndices().size();
994
995 if (nb_row_dofs && nb_col_dofs) {
996
997 const bool diag = (row_data.getFieldEntities()[0]->getLocalUniqueId() ==
998 col_data.getFieldEntities()[0]->getLocalUniqueId());
999
1000 FTensor::Index<'i', SPACE_DIM> i;
1001 FTensor::Index<'j', SPACE_DIM> j;
1002 FTensor::Index<'k', SPACE_DIM> k;
1003 FTensor::Index<'l', SPACE_DIM> l;
1004
1005 // get element volume
1006 double vol = OpBase::getMeasure();
1007
1008 // get intergrayion weights
1009 auto t_w = OpBase::getFTensor0IntegrationWeight();
1010
1011 // get derivatives of base functions on rows
1012 auto t_row_diff_base = row_data.getFTensor1DiffN<SPACE_DIM>();
1013
1014 // Elastic stiffness tensor (4th rank tensor with minor and major
1015 // symmetry)
1016 auto t_D = getFTensor4DdgFromMat<SPACE_DIM, SPACE_DIM, S>(*matD);
1017
1018 // get coordinate at integration points
1019 auto t_coords = OpBase::getFTensor1CoordsAtGaussPts();
1020
1021 // iterate over integration points
1022 for (int gg = 0; gg != OpBase::nbIntegrationPts; ++gg) {
1023
1024 // calculate scalar weight times element volume
1025 double a = t_w * vol * betaCoeff(t_coords(0), t_coords(1), t_coords(2));
1026
1027 // iterate over row base functions
1028 int rr = 0;
1029 for (; rr != OpBase::nbRows / SPACE_DIM; ++rr) {
1030
1031 // get sub matrix for the row
1032 auto t_m = OpBase::template getLocMat<SPACE_DIM>(SPACE_DIM * rr);
1033
1034 FTensor::Christof<double, SPACE_DIM, SPACE_DIM> t_rowD;
1035 // I mix up the indices here so that it behaves like a
1036 // Dg. That way I don't have to have a separate wrapper
1037 // class Christof_Expr, which simplifies things.
1038 t_rowD(l, j, k) = t_D(i, j, k, l) * (a * t_row_diff_base(i));
1039
1040 // get derivatives of base functions for columns
1041 auto t_col_diff_base = col_data.getFTensor1DiffN<SPACE_DIM>(gg, 0);
1042
1043 // iterate column base functions
1044 const auto nb_cols = (diag) ? rr : OpBase::nbCols / SPACE_DIM - 1;
1045 for (int cc = 0; cc <= nb_cols; ++cc) {
1046
1047 // integrate block local stiffens matrix
1048 t_m(i, j) += t_rowD(i, j, k) * t_col_diff_base(k);
1049
1050 // move to next column base function
1051 ++t_col_diff_base;
1052
1053 // move to next block of local stiffens matrix
1054 ++t_m;
1055 }
1056
1057 // move to next row base function
1058 ++t_row_diff_base;
1059 }
1060
1061 for (; rr < OpBase::nbRowBaseFunctions; ++rr)
1062 ++t_row_diff_base;
1063
1064 // move to next integration weight
1065 ++t_w;
1066 ++t_D;
1067 ++t_coords;
1068 }
1069
1070 // Copy symmetry
1071 if (diag) {
1072 for (int rr = 0; rr != OpBase::nbRows / SPACE_DIM - 1; ++rr) {
1073 auto t_m_rr = getFTensor2FromArray<SPACE_DIM, SPACE_DIM, SPACE_DIM>(
1074 this->locMat, SPACE_DIM * rr, SPACE_DIM * (rr + 1));
1075 auto t_m_cc = getFTensor2FromArray<SPACE_DIM, SPACE_DIM>(
1076 this->locMat, SPACE_DIM * (rr + 1), SPACE_DIM * rr,
1077 SPACE_DIM * OpBase::nbCols);
1078 for (int cc = rr + 1; cc < OpBase::nbCols / SPACE_DIM; ++cc) {
1079 t_m_rr(i, j) = t_m_cc(j, i);
1080 ++t_m_rr;
1081 ++t_m_cc;
1082 }
1083 }
1084 }
1085
1086 }
1087
1088 MoFEMFunctionReturn(0);
1089}
