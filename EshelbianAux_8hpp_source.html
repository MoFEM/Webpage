138template <int DIM> inline auto get_uniq_nb(double *ptr) {
139 std::array<double, DIM> tmp;
140 std::copy(ptr, &ptr[DIM], tmp.begin());
141 std::sort(tmp.begin(), tmp.end());
142 isEq::absMax = std::max(std::abs(tmp[0]), std::abs(tmp[DIM - 1]));
143 constexpr double eps = std::numeric_limits<double>::epsilon();
144 isEq::absMax = std::max(isEq::absMax, static_cast<double>(eps));
145 return std::distance(tmp.begin(), std::unique(tmp.begin(), tmp.end(), is_eq));
