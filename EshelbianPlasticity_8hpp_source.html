419  MoFEMFunctionBegin;
420 
421  if (assembleSymmetry) {
422  const auto row_nb_dofs = row_data.getIndices().size();
423  const auto col_nb_dofs = col_data.getIndices().size();
424  transposeK.resize(col_nb_dofs, row_nb_dofs, false);
425  noalias(transposeK) = trans(K);

427  }
428 
429  CHKERR MatSetValues<AssemblyTypeSelector<A>>(this->getTSB(), row_data,
430  col_data, K, ADD_VALUES);
431  if (assembleSymmetry) {
432  CHKERR MatSetValues<AssemblyTypeSelector<A>>(
433  this->getTSB(), col_data, row_data, transposeK, ADD_VALUES);
434  }
435 
436  MoFEMFunctionReturn(0);
437  }

442  MoFEMFunctionReturnHot(0);

444  nF.clear();
