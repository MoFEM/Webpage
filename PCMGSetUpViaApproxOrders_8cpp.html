589PCMGSetUpViaApproxOrdersCtx::buildProlongationOperator(bool use_mat_a,
590 int verb) {
591 MoFEMFunctionBegin;
592 verb = verb > verboseLevel ? verb : verboseLevel;
593
594 MPI_Comm comm;
595 CHKERR PetscObjectGetComm((PetscObject)dM, &comm);
596
597 MOFEM_LOG_C("PCMGViaApproximationOrders", Sev::inform, "set MG levels %u\n",
598 nbLevels);
599
600 MOFEM_LOG_CHANNEL("SYNC");
601 MOFEM_LOG_TAG("SYNC", "PCMGViaApproximationOrders")
602
603 std::vector<IS> is_vec(nbLevels + 1);
604 std::vector<int> is_glob_size(nbLevels + 1), is_loc_size(nbLevels + 1);
605
606 for (int kk = 0; kk < nbLevels; kk++) {
607
608 // get indices up to up to give approximation order
609 CHKERR createIsAtLevel(kk, &is_vec[kk]);
610 CHKERR ISGetSize(is_vec[kk], &is_glob_size[kk]);
611 CHKERR ISGetLocalSize(is_vec[kk], &is_loc_size[kk]);
612
613 MOFEM_LOG_C("SYNC", Sev::inform,
614 "Nb. dofs at level [ %d ] global %u local %d\n", kk,
615 is_glob_size[kk], is_loc_size[kk]);
616
617 // if no dofs on level kk finish here
618 if (is_glob_size[kk] == 0) {
619 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY, "no dofs at level");
620 }
621 }
622
623 for (int kk = 0; kk != nbLevels; kk++) {
624 if (kk == nbLevels - 1 && use_mat_a) {
625 CHKERR DMMGViaApproxOrdersPushBackCoarseningIS(dM, is_vec[kk], A, false,
626 false);
627 } else {
628 if (kk > 0) {
629 // Not coarse level
630 CHKERR DMMGViaApproxOrdersPushBackCoarseningIS(dM, is_vec[kk], A, true,
631 shellSubA);
632 } else {
633 // Coarse lave is compressed matrix allowing for factorization when
634 // needed
635 CHKERR DMMGViaApproxOrdersPushBackCoarseningIS(dM, is_vec[kk], A, true,
636 false);
637 }
638 }
639 }
640
641 for (unsigned int kk = 0; kk < is_vec.size(); kk++) {
642 CHKERR destroyIsAtLevel(kk, &is_vec[kk]);
643 }
644
645 MOFEM_LOG_SEVERITY_SYNC(comm, Sev::inform);
646 MOFEM_LOG_CHANNEL("SYNC");
647
648 MoFEMFunctionReturn(0);
649}
