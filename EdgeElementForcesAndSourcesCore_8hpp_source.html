269  auto t_normal = getFTensor1FromMat<3>(normals_at_gauss_pts);
270  auto t_dir = getFTensor1TangentAtGaussPts<3>();
271  for(auto gg = 0; gg!=getGaussPts().size2(); ++gg) {
272  t_normal(i) = FTensor::levi_civita(i, j, k) * t_dir(j) * vec(k);
