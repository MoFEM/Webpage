116 Tensor4(T *d, const int shift, const int i = 1) : inc(i)
117 {
118 int s = 0;
119 for (int i = 0; i != Tensor_Dim0; ++i)
120 {
121 for (int j = 0; j != Tensor_Dim1; ++j)
122 {
123 for (int k = 0; k != Tensor_Dim2; ++k)
124 {
125 for (int l = 0; l != Tensor_Dim3; ++l)
126 {
127 data[i][j][k][l] = &d[s];
128 s += shift;
129 }
130 }
131 }
132 }
