97 const int nb_integration_pts = getGaussPts().size2();
98 auto t_w = getFTensor0IntegrationWeight();

109 double error = 0;

118 nf[r] += alpha * t_row_base * diff;

120 }

123 ++t_val;
124 ++t_coords;
125 }
126

128 CHKERR VecSetValue(commonDataPtr->L2Vec, index, error, ADD_VALUES);
