195 CHKERR VecGhostUpdateBegin(arcPtrRaw->ghostDiag, INSERT_VALUES,
196 SCATTER_FORWARD);
197 CHKERR VecGhostUpdateEnd(arcPtrRaw->ghostDiag, INSERT_VALUES,
198 SCATTER_FORWARD);

201 } break;

203 CHKERR calculateDxAndDlambda();
204 CHKERR calculateDb();
205 if (arcPtrRaw->getPart() == arcPtrRaw->mField.get_comm_rank()) {
206 int idx = arcPtrRaw->getPetscGlobalDofIdx();
207 CHKERR MatSetValue(ts_B, idx, idx, 1, ADD_VALUES);
208 }
209 CHKERR MatAssemblyBegin(ts_B, MAT_FLUSH_ASSEMBLY);
210 CHKERR MatAssemblyEnd(ts_B, MAT_FLUSH_ASSEMBLY);
211 } break;
212 default:
213 break;
214 }

222 }

240 CHKERR VecGhostUpdateEnd(arcPtrRaw->dx, INSERT_VALUES, SCATTER_FORWARD);

244 double *array;
245 CHKERR VecGetArray(ts_u, &array);

247 CHKERR VecRestoreArray(ts_u, &array);
248 CHKERR VecGetArray(arcPtrRaw->x0, &array);
249 arcPtrRaw->dLambda -= array[idx];
250 CHKERR VecRestoreArray(arcPtrRaw->x0, &array);
251 CHKERR VecGetArray(arcPtrRaw->dx, &array);
252 array[idx] = 0;
253 CHKERR VecRestoreArray(arcPtrRaw->dx, &array);

255 CHKERR VecGhostUpdateBegin(arcPtrRaw->ghosTdLambda, INSERT_VALUES,
256 SCATTER_FORWARD);
257 CHKERR VecGhostUpdateEnd(arcPtrRaw->ghosTdLambda, INSERT_VALUES,

266 arcPtrRaw->dLambda, arcPtrRaw->dx2, nrm2_u_t);

277PetscErrorCode alpha_adapt(TS ts, PetscReal t, Vec X, Vec Xdot,
278 PetscReal *nextdt, PetscBool *ok, void *ctx) {
279 PetscFunctionBegin;
280
281 SNES snes;
282 SNESConvergedReason snesreason;
283 int snesits;
284 int its_d = 6;
285 double gamma = 0.5;
286 double min_dt = 0.01;
287 double max_dt = 10.0;
288 double atol, rtol, stol;
289 int maxit, maxf;
290 double reduction;
291
292 CHKERR TSGetSNES(ts, &snes);
293 CHKERR SNESGetTolerances(snes, &atol, &rtol, &stol, &maxit, &maxf);
294 AlphaAdaptCtx *alpha_adapt_ctx = (AlphaAdaptCtx *)(ctx);
295 ArcLengthCtx *arc_ptr = alpha_adapt_ctx->arcPtrRaw;
296 double &maxnorm = alpha_adapt_ctx->maxnorm;
297
298 try {
299
300 po::options_description config_file_options;
301 config_file_options.add_options()(
302 "its_d", po::value<int>(&its_d)->default_value(its_d));
303 config_file_options.add_options()(
304 "gamma", po::value<double>(&gamma)->default_value(gamma));
305 config_file_options.add_options()(
306 "min_dt", po::value<double>(&min_dt)->default_value(min_dt));
307 config_file_options.add_options()(
308 "max_dt", po::value<double>(&max_dt)->default_value(max_dt));
309
310 config_file_options.add_options()(
311 "snes.atol", po::value<double>(&atol)->default_value(atol));
312 config_file_options.add_options()(
313 "snes.rtol", po::value<double>(&rtol)->default_value(rtol));
314 config_file_options.add_options()(
315 "snes.stol", po::value<double>(&stol)->default_value(stol));
316 config_file_options.add_options()(
317 "snes.maxit", po::value<int>(&maxit)->default_value(maxit));
318 config_file_options.add_options()(
319 "snes.maxnorm", po::value<double>(&maxnorm)->default_value(maxnorm));
320
321 config_file_options.add_options()(
322 "arc_length.alpha",
323 po::value<double>(&arc_ptr->alpha)->default_value(arc_ptr->alpha));
324 config_file_options.add_options()(
325 "arc_length.beta",
326 po::value<double>(&arc_ptr->beta)->default_value(arc_ptr->beta));
327 config_file_options.add_options()(
328 "arc_length.step_size",
329 po::value<double>(&arc_ptr->s)->default_value(arc_ptr->s));
330
331 config_file_options.add_options()(
332 "numerical_damping",
333 po::value<double>(alpha_adapt_ctx->damping_ptr)
334 ->default_value(*alpha_adapt_ctx->damping_ptr));
335
336 ifstream file("snes_config.in");
337 po::parsed_options parsed =
338 parse_config_file(file, config_file_options, false);
339 po::variables_map vm;
340 store(parsed, vm);
341 po::notify(vm);
342
343 } catch (exception &ex) {
344 SETERRQ(PETSC_COMM_SELF, MOFEM_STD_EXCEPTION_THROW,
345 "error parsing material elastic configuration file");
346 }
347
348 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### its_d %d\n", its_d);
349 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### gamma %3.2f\n", gamma);
350 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### min_dt %3.2f\n", min_dt);
351 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### max_dt %3.2f\n", max_dt);
352
353 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### snes.atol %4.2e\n", atol);
354 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### snes.rtol %4.2e\n", rtol);
355 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### snes.stol %4.2e\n", stol);
356 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### snes.maxit %d\n", maxit);
357
358 CHKERR PetscPrintf(PETSC_COMM_WORLD, "### arc_length.alpha %4.2e\n",

366 *alpha_adapt_ctx->damping_ptr);
367
368 CHKERR SNESSetTolerances(snes, atol, rtol, stol, maxit, maxf);
369
370 CHKERR SNESGetConvergedReason(snes, &snesreason);
371 if (snesreason < 0) {
372 *ok = PETSC_FALSE;
373 *nextdt *= 0.1;
374 goto finally;
375 }
376
377 CHKERR SNESGetIterationNumber(snes, &snesits);
378 reduction = pow((double)its_d / ((double)(snesits + 1)), gamma);
379
380 *ok = PETSC_TRUE;
381 *nextdt *= reduction;
382
383finally:
384
385 *nextdt = PetscMax(*nextdt, min_dt);
386 *nextdt = PetscMin(*nextdt, max_dt);
387
388 PetscFunctionReturn(0);
389}

401PetscErrorCode monitor(TS ts, PetscInt step, PetscReal crtime, Vec u,
402 void *ctx) {
403 PetscFunctionBegin;
404 MonitorCtx *monitor_ctx = (MonitorCtx *)(ctx);
405 DM dm = monitor_ctx->dm;
406 MoFEM::Interface *m_field_ptr = monitor_ctx->m_field_ptr;
407 ArcLengthCtx *arc_ctx = monitor_ctx->arc_ctx;
408 SolidShellPrismElement::PostProcFatPrismOnTriangleOnRefinedMesh
409 *shell_post_proc_ptr = monitor_ctx->shell_post_proc_ptr;
410 PostProcFatPrismOnRefinedMesh *solid_post_proc_ptr =
411 monitor_ctx->solid_post_proc_ptr;
412
413 CHKERR DMoFEMMeshToGlobalVector(dm, u, INSERT_VALUES, SCATTER_REVERSE);
414 PetscPrintf(PETSC_COMM_WORLD, "Post-processing data\n");
415 CHKERR DMoFEMLoopFiniteElements(dm, "SHELL", shell_post_proc_ptr);
416 CHKERR DMoFEMLoopFiniteElements(dm, "SHELL", solid_post_proc_ptr);
417 // Save data on mesh
418 PetscPrintf(PETSC_COMM_WORLD, "Writing file ...\n");
419 {
420 ostringstream o1;
421 o1 << "shell_" << step << ".h5m";
422 rval = shell_post_proc_ptr->postProcMesh.write_file(
423 o1.str().c_str(), "MOAB", "PARALLEL=WRITE_PART");
424 CHKERRQ_MOAB(rval);
425 }
426 {
427 ostringstream o1;
428 o1 << "solid_" << step << ".h5m";
429 rval = solid_post_proc_ptr->postProcMesh.write_file(
430 o1.str().c_str(), "MOAB", "PARALLEL=WRITE_PART");
431 CHKERRQ_MOAB(rval);
432 }
433
434 // print displacement
435 {
436 const DofEntity_multiIndex *dofs_ptr;
437 CHKERR m_field_ptr->get_dofs(&dofs_ptr);
438 for (_IT_CUBITMESHSETS_BY_NAME_FOR_LOOP_((*m_field_ptr), "RESULTS", it)) {
439 EntityHandle meshset = it->getMeshset();
440 Range nodes;
441 rval = m_field_ptr->get_moab().get_entities_by_type(meshset, MBVERTEX,
442 nodes, true);
443 CHKERRQ_MOAB(rval);
444 Range::iterator nit = nodes.begin();
445 for (; nit != nodes.end(); nit++) {
446 DofEntityByNameAndEnt::iterator dit, hi_dit;
447 dit = dofs_ptr->get<Composite_Name_And_Ent_mi_tag>().lower_bound(
448 boost::make_tuple("DISPLACEMENT", *nit));
449 hi_dit = dofs_ptr->get<Composite_Name_And_Ent_mi_tag>().upper_bound(
450 boost::make_tuple("DISPLACEMENT", *nit));
451 for (; dit != hi_dit; dit++) {
452 EntityHandle ent = dit->get()->getEnt();
453 double coords[3];
454 rval = m_field_ptr->get_moab().get_coords(&*nit, 1, coords);
455 CHKERRQ_MOAB(rval);
456 PetscPrintf(PETSC_COMM_WORLD,
457 "STEP %d LAMBDA %6.4e DISPLACEMENT [ %d ] %6.4e coord [ "
458 "%3.4f %3.4f %3.4f ] EntityHandle %ld\n",
459 step, arc_ctx->getFieldData(),
460 dit->get()->getDofCoeffIdx(), dit->get()->getFieldData(),
461 coords[0], coords[1], coords[2], ent);
462 }
463 }
464 }
465 }
466
467 if (step) {
468 CHKERR VecWAXPY(arc_ctx->dx, -1, arc_ctx->x0, u);
469 double dot0;
470 CHKERR VecDot(monitor_ctx->bSign, arc_ctx->dx, &dot0);
471 double dot1;
472 CHKERR VecDot(arc_ctx->dx, arc_ctx->dx, &dot1);
473 const double eps = 1e-12;
474 if (fabs(dot1 / dot0) > eps) {
475 CHKERR VecCopy(arc_ctx->dx, monitor_ctx->bSign);
476 {
477 const MoFEM::Problem *problem_ptr;
478 CHKERR DMMoFEMGetProblemPtr(dm, &problem_ptr);
479 double *array;
480 CHKERR VecGetArray(monitor_ctx->bSign, &array);
481 const int max_order = 1; // max controlled dof order
482 for (_IT_NUMEREDDOF_ROW_FOR_LOOP_(problem_ptr, dof)) {
483 int idx = dof->get()->getPetscLocalDofIdx();
484 if (idx != -1 && dof->get()->getDofOrder() > max_order) {
485 // zero higher-order dofs
486 array[idx] = 0;
487 }
488 }
489 CHKERR VecRestoreArray(monitor_ctx->bSign, &array);
490 }
491 CHKERR VecDot(monitor_ctx->bSign, arc_ctx->dx, &dot1);
492 double nrm2 = sqrt(fabs(dot1));
493 // \alpha_1 = \alpha_0 \| dx \| \frac{b \cdot dx}{dx \cdot dx}
494 double alpha = fabs(arc_ctx->alpha * dot0 / (dot1 / nrm2));
495 arc_ctx->setAlphaBeta(alpha, arc_ctx->beta);
496 CHKERR VecScale(monitor_ctx->bSign, 1. / nrm2);
497 }
498 CHKERR VecCopy(u, arc_ctx->x0);
499 }
500
501 // TS_Alpha *th = (TS_Alpha*)ts->data;
502 // CHKERR VecView(th->V0,PETSC_VIEWER_STDOUT_WORLD); CHKERRQ(ierr);
503
504 PetscFunctionReturn(0);
505}
