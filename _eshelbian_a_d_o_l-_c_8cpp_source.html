199 MoFEMErrorCode recordTape(const int tape, DTensor2Ptr *t_h_ptr) {
200 MoFEMFunctionBegin;
201 FTensor::Index<'i', 3> i;
202 FTensor::Index<'j', 3> j;
203 FTensor::Index<'I', 3> I;
204 FTensor::Index<'J', 3> J;
205 FTensor::Number<0> N0;
206 FTensor::Number<1> N1;
207 FTensor::Number<2> N2;
208
209 CHKERR getOptions();
210
211 auto t_kd = FTensor::Kronecker_Delta_symmetric<int>();
212
213 auto ih = get_h();
214 // auto iH = get_H();
215 if (t_h_ptr)
216 ih(i, j) = (*t_h_ptr)(i, j);
217 else {
218 ih(i, j) = t_kd(i, j);
219 }
220
221 auto r_P = get_P();
222
223 enableMinMaxUsingAbs();
224 trace_on(tape);
225
226 // Set active variables to ADOL-C
227 th(i, j) <<= get_h()(i, j);
228
229 tF(i, I) = th(i, I);
230 CHKERR determinantTensor3by3(tF, detF);
231 CHKERR invertTensor3by3(tF, detF, tInvF);
232
233 // Deformation and strain
234 tC(I, J) = tF(i, I) * tF(i, J);
235 tE(I, J) = tC(I, J);
236 tE(N0, N0) -= 1;
237 tE(N1, N1) -= 1;
238 tE(N2, N2) -= 1;
239 tE(I, J) *= 0.5;
240
241 // Energy
242 trE = tE(I, I);
243 energy = 0.5 * lambda * trE * trE + mu * (tE(I, J) * tE(I, J));
244
245 // Stress Piola II
246 tS(I, J) = tE(I, J);
247 tS(I, J) *= 2 * mu;
248 tS(N0, N0) += lambda * trE;
249 tS(N1, N1) += lambda * trE;
250 tS(N2, N2) += lambda * trE;
251 // Stress Piola I
252 tP(i, J) = tF(i, I) * tS(I, J);
253
254 // Set dependent variables to ADOL-C
255 tP(i, j) >>= r_P(i, j);
256
257 trace_off();
258
259 MoFEMFunctionReturn(0);
260 }

475 MoFEMErrorCode getMatDPtr(boost::shared_ptr<MatrixDouble> mat_D_ptr,
476 double bulk_modulus_K, double shear_modulus_G) {
477 MoFEMFunctionBegin;
478 //! [Calculate elasticity tensor]
479 auto set_material_stiffness = [&]() {
480 FTensor::Index<'i', SPACE_DIM> i;
481 FTensor::Index<'j', SPACE_DIM> j;
482 FTensor::Index<'k', SPACE_DIM> k;
483 FTensor::Index<'l', SPACE_DIM> l;
484 constexpr auto t_kd = FTensor::Kronecker_Delta_symmetric<int>();
485 double A = (SPACE_DIM == 2)
486 ? 2 * shear_modulus_G /
487 (bulk_modulus_K + (4. / 3.) * shear_modulus_G)
488 : 1;
489 auto t_D = getFTensor4DdgFromMat<SPACE_DIM, SPACE_DIM, 0>(*mat_D_ptr);
490 t_D(i, j, k, l) = 2 * shear_modulus_G * ((t_kd(i, k) ^ t_kd(j, l)) / 4.) +
491 A * (bulk_modulus_K - (2. / 3.) * shear_modulus_G) *
492 t_kd(i, j) * t_kd(k, l);
493 };
494 //! [Calculate elasticity tensor]
495 constexpr auto size_symm = (SPACE_DIM * (SPACE_DIM + 1)) / 2;
496 mat_D_ptr->resize(size_symm * size_symm, 1);
497 set_material_stiffness();
498 MoFEMFunctionReturn(0);
