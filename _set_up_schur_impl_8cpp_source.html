42 auto get_ents_by_dim = [&](int dim) {
43 Range ents;
44 CHKERR mField.get_moab().get_entities_by_dimension(0, dim, ents);
45 return ents;

47
48 std::vector<std::string> schur_field_list{
49 epCorePtr->StretchTensor, epCorePtr->bubbleField, epCorePtr->rotAxis,
50 epCorePtr->spatialL2Disp, epCorePtr->piolaStress};

53 auto create_schur_dm = [&](SmartPetscObj<DM> &dm_sub) {
54 MoFEMFunctionBegin;
55
56 dm_sub = createDM(mField.get_comm(), "DMMOFEM");
57 CHKERR DMMoFEMCreateSubDM(dm_sub, epCorePtr->dmElastic, "SUB_SCHUR");
58 CHKERR DMMoFEMSetSquareProblem(dm_sub, PETSC_TRUE);
59 CHKERR DMMoFEMSetIsPartitioned(dm_sub, PETSC_TRUE);
60 CHKERR DMMoFEMAddElement(dm_sub, epCorePtr->elementVolumeName);
61 CHKERR DMMoFEMAddElement(dm_sub, epCorePtr->naturalBcElement);
62
63 auto faces_ptr = boost::make_shared<Range>(get_ents_by_dim(SPACE_DIM - 1));
64 std::vector<boost::shared_ptr<Range>> dm_range_list{faces_ptr};
65
66 if (field_list.size() != dm_range_list.size()) {
67 SETERRQ(PETSC_COMM_WORLD, MOFEM_DATA_INCONSISTENCY,
68 "Both ranges should have the same size");
69 }
70
71 int r_idx = 0;
72 for (auto f : field_list) {
73 MOFEM_LOG("EP", Sev::inform) << "Add schur field: " << f;
74 CHKERR DMMoFEMAddSubFieldRow(dm_sub, f.c_str(), dm_range_list[r_idx]);
75 CHKERR DMMoFEMAddSubFieldCol(dm_sub, f.c_str(), dm_range_list[r_idx]);
76 ++r_idx;
77 }

79 MoFEMFunctionReturn(0);

83 MoFEMFunctionBegin;

85 std::vector<SmartPetscObj<IS>> is_vec;
86 std::vector<Range *> range_list_ptr(schur_field_list.size(), nullptr);
87 range_list_ptr.back() = &vols;
88 int r_idx = 0;
89 for (auto f : schur_field_list) {
90 SmartPetscObj<IS> is;
91 CHKERR mField.getInterface<ISManager>()->isCreateProblemFieldAndRank(
92 "ELASTIC_PROBLEM", ROW, f, 0, MAX_DOFS_ON_ENTITY, is,
93 range_list_ptr[r_idx]);
94 is_vec.push_back(is);
95 ++r_idx;
96 }
97 if (is_vec.size()) {
98 is_a00 = is_vec[0];
99 for (auto i = 1; i < is_vec.size(); ++i) {
100 IS is_union_raw;
101 CHKERR ISExpand(is_a00, is_vec[i], &is_union_raw);
102 is_a00 = SmartPetscObj<IS>(is_union_raw);
103 }
104 CHKERR ISSort(is_a00);
105 }
106 MoFEMFunctionReturn(0);
107 };

110 CHKERR KSPSetFromOptions(solver);

112 PetscBool is_pcfs = PETSC_FALSE;
113 PetscObjectTypeCompare((PetscObject)pc, PCFIELDSPLIT, &is_pcfs);
114 if (is_pcfs) {

120 if (S) {

129 auto set_ops = [&]() {
130 MoFEMFunctionBegin;
131 std::vector<boost::shared_ptr<Range>> ranges_list(schur_field_list.size(),
132 nullptr);
133 ranges_list.back() =
134 boost::make_shared<Range>(get_ents_by_dim(SPACE_DIM));
135 std::vector<SmartPetscObj<AO>> ao_list(schur_field_list.size(),
136 SmartPetscObj<AO>());
137 std::vector<SmartPetscObj<Mat>> mat_list(schur_field_list.size(),
138 SmartPetscObj<Mat>());
139 std::vector<bool> symm_list(schur_field_list.size(), false);
140
141 auto dm_is = getDMSubData(schur_dm)->getSmartRowIs();
142 aoUp = createAOMappingIS(dm_is, PETSC_NULL);
143 ao_list.back() = aoUp;
144 mat_list.back() = S;
145
146 const bool symmetric_system =
147 EshelbianCore::gradApperoximator <= MODERATE_ROT;
148 if (!symmetric_system) {
149
150 epCorePtr->elasticFeLhs->getOpPtrVector().push_front(
151 new OpSchurAssembleBegin());
152 epCorePtr->elasticFeLhs->getOpPtrVector().push_back(

159 new OpSchurAssembleEnd<SCHUR_DSYSV>(schur_field_list, ranges_list,
160 ao_list, mat_list, symm_list,
161 true));
162
163 } else {
164
165 epCorePtr->elasticFeLhs->getOpPtrVector().push_front(
166 new OpSchurAssembleBegin());
167 epCorePtr->elasticFeLhs->getOpPtrVector().push_back(
168 new OpSchurAssembleEnd<SCHUR_DGESV>(schur_field_list, ranges_list,
169 ao_list, mat_list, symm_list,
170 false));
171 epCorePtr->elasticBcLhs->getOpPtrVector().push_front(
172 new OpSchurAssembleBegin());
173 epCorePtr->elasticBcLhs->getOpPtrVector().push_back(
174 new OpSchurAssembleEnd<SCHUR_DGESV>(schur_field_list, ranges_list,
175 ao_list, mat_list, symm_list,
176 false));
177 }
178 MoFEMFunctionReturn(0);
179 };

182 MoFEMFunctionBegin;
