490 // Store edge nodes coordinates in FTensor

492 FTensor::Tensor1<double *, 3> t_node_edge[2] = {
493 FTensor::Tensor1<double *, 3>(edge_node_coords, &edge_node_coords[1],
494 &edge_node_coords[2]),
495 FTensor::Tensor1<double *, 3>(&edge_node_coords[3],
496 &edge_node_coords[4],
497 &edge_node_coords[5])};
498 FTensor::Index<'i', 3> i;
499 double max_edge_l = 0;
500 for (auto edge : edges) {
501 int num_nodes;
502 const EntityHandle *conn;
503 CHKERR moab.get_connectivity(edge, conn, num_nodes, true);
504 CHKERR moab.get_coords(conn, num_nodes, edge_node_coords);
505 t_node_edge[0](i) -= t_node_edge[1](i);
506 double l = sqrt(t_node_edge[0](i) * t_node_edge[0](i));
507 max_edge_l = (max_edge_l < l) ? l : max_edge_l;
508 }
509 CHKERR VecSetValue(length_vec, m_field.get_comm_rank(), max_edge_l, INSERT_VALUES);
