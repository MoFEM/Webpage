1033  t_C(1, 1, 2, 2) = -c;
1034 
1035  t_C(2, 2, 0, 0) = -c;
1036  t_C(2, 2, 1, 1) = -c;
1037  t_C(2, 2, 2, 2) = 1. / young_modulus;
1038

1040  t_C(0, 1, 0, 1) = d;
1041  t_C(0, 2, 0, 2) = d;
1042  t_C(1, 2, 1, 2) = d;
