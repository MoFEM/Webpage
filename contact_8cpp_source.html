199 MoFEMFunctionBegin;
200 Simple *simple = mField.getInterface<Simple>();

202 CHKERR PetscOptionsGetInt(PETSC_NULL, "", "-order", &order, PETSC_NULL);
203 CHKERR PetscOptionsGetInt(PETSC_NULL, "", "-geom_order", &geom_order,
204 PETSC_NULL);
205 MOFEM_LOG("CONTACT", Sev::inform) << "Order " << order;
206 MOFEM_LOG("CONTACT", Sev::inform) << "Geom order " << geom_order;
207
208 // Select base
209 enum bases { AINSWORTH, DEMKOWICZ, LASBASETOPT };
210 const char *list_bases[LASBASETOPT] = {"ainsworth", "demkowicz"};
211 PetscInt choice_base_value = AINSWORTH;
212 CHKERR PetscOptionsGetEList(PETSC_NULL, NULL, "-base", list_bases,
213 LASBASETOPT, &choice_base_value, PETSC_NULL);
214
215 FieldApproximationBase base;
216 switch (choice_base_value) {
217 case AINSWORTH:
218 base = AINSWORTH_LEGENDRE_BASE;
219 MOFEM_LOG("CONTACT", Sev::inform)
220 << "Set AINSWORTH_LEGENDRE_BASE for displacements";
221 break;
222 case DEMKOWICZ:
223 base = DEMKOWICZ_JACOBI_BASE;
224 MOFEM_LOG("CONTACT", Sev::inform)
225 << "Set DEMKOWICZ_JACOBI_BASE for displacements";
226 break;
227 default:
228 base = LASTBASE;
229 break;
230 }
231
232 // Note: For tets we have only H1 Ainsworth base, for Hex we have only H1
233 // Demkowicz base. We need to implement Demkowicz H1 base on tet.
234 CHKERR simple->addDomainField("U", H1, base, SPACE_DIM);
235 CHKERR simple->addBoundaryField("U", H1, base, SPACE_DIM);
236 CHKERR simple->addDomainField("SIGMA", CONTACT_SPACE, DEMKOWICZ_JACOBI_BASE,
237 SPACE_DIM);
238 CHKERR simple->addBoundaryField("SIGMA", CONTACT_SPACE, DEMKOWICZ_JACOBI_BASE,
239 SPACE_DIM);
240 CHKERR simple->addDataField("GEOMETRY", H1, base, SPACE_DIM);

242 CHKERR simple->setFieldOrder("U", order);
243 CHKERR simple->setFieldOrder("GEOMETRY", geom_order);

254 auto filter_blocks = [&](auto skin) {
255 bool is_contact_block = false;
256 Range contact_range;
257 for (auto m :
258 mField.getInterface<MeshsetsManager>()->getCubitMeshsetPtr(std::regex(
259
260 (boost::format("%s(.*)") % "CONTACT").str()
261
262 ))
263
264 ) {
265 is_contact_block =
266 true; ///< bloks interation is collectibe, so that is set irrespective
267 ///< if there are enerities in given rank or not in the block
268 MOFEM_LOG("CONTACT", Sev::inform)
269 << "Find contact block set: " << m->getName();
270 auto meshset = m->getMeshset();
271 Range contact_meshset_range;
272 CHKERR mField.get_moab().get_entities_by_dimension(
273 meshset, SPACE_DIM - 1, contact_meshset_range, true);
274
275 CHKERR mField.getInterface<CommInterface>()->synchroniseEntities(
276 contact_meshset_range);
277 contact_range.merge(contact_meshset_range);
278 }
279 if (is_contact_block) {
280 MOFEM_LOG("SYNC", Sev::inform)
281 << "Nb entities in contact surface: " << contact_range.size();
282 MOFEM_LOG_SYNCHRONISE(mField.get_comm());
283 skin = intersect(skin, contact_range);
284 }
285 return skin;
286 };
