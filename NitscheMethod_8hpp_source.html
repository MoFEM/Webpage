178  commonData.nbTetGaussPts++;

182  0.5 * getNormalsAtGaussPtss();
183  commonData.hoCoordsAtGaussPts.resize(4);
184  commonData.hoCoordsAtGaussPts[faceInRespectToTet] =
185  getCoordsAtGaussPts();
