31MoFEMErrorCode TimeScale::timeData(std::string delimiter) {
32 MoFEMFunctionBegin;
33 PetscBool arg_found = PETSC_FALSE;
34 char time_file_name[255] = {'\0'};
35 CHKERR PetscOptionsGetString(PETSC_NULL, PETSC_NULL, fileNameFlag.c_str(),
36 time_file_name, 255, &arg_found);
37 if (arg_found) {
38 fileName = std::string(time_file_name);
39 }
40 if (!arg_found && fileName.empty() && errorIfFileNotGiven) {
41 SETERRQ1(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
42 "*** ERROR %s (time_data FILE NEEDED)", fileName.c_str());
43 } else if (!arg_found && fileName.empty() && !errorIfFileNotGiven) {
44 MOFEM_LOG_C("WORLD", Sev::warning,
45 "The %s file not provided. Loading scaled with time.",
46 fileName.c_str());
47 scalingMethod = [this](double time) { return this->getLinearScale(time); };
48 MoFEMFunctionReturnHot(0);
49 }
50
51 std::ifstream in_file_stream(fileName);
52 if (!in_file_stream.is_open() && errorIfFileNotGiven) {
53 SETERRQ1(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
54 "*** ERROR data file < %s > open unsuccessful", fileName.c_str());
55 } else if (!in_file_stream.is_open() && !errorIfFileNotGiven) {
56 MOFEM_LOG("WORLD", Sev::warning)
57 << "*** Warning data file " << fileName
58 << " open unsuccessful. Using linear time scaling.";
59 scalingMethod = [this](double time) { return this->getLinearScale(time); };
60 MoFEMFunctionReturnHot(0);
61 }
62
63 in_file_stream.seekg(0);
64 std::string line;
65 double time = 0.0, value = 0.0;
66 tSeries[time] = value;
67
68 std::regex rgx(delimiter.c_str());
69 std::sregex_token_iterator end;
70 while (std::getline(in_file_stream, line)) {
71 std::sregex_token_iterator iter(line.begin(), line.end(), rgx, -1);
72 auto time_str = iter;
73 auto value_str = ++iter;
74 if (time_str == end || value_str == end) {
75 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
76 "*** ERROR read data file error (check input time data file) ");
77 }
78 time = std::stod(time_str->str());
79 value = std::stod(value_str->str());
80 tSeries[time] = value;
81 }
82 in_file_stream.close();
83 scalingMethod = [this](double time) { return this->getScaleFromData(time); };
84 if (in_file_stream.is_open()) {
85 SETERRQ(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
86 "*** ERROR file close unsuccessful");
87 }
88 MoFEMFunctionReturn(0);
89}

93double TimeScale::getScaleFromData(const double time) {
94 if (tSeries.empty())
95 return 0.;
96
97 auto it = tSeries.find(time);
98 if (it != tSeries.end()) {
99 return it->second;
100 } else {
101 auto it = tSeries.lower_bound(time);
102 if (it == tSeries.end()) {
103 return (--it)->second;
104 }
105 if (it == tSeries.begin()) {
106 return it->second;
107 }
108 auto upper = *(it);
109 it--;
110 if (it == tSeries.end()) {
111 return upper.second;
112 }
113 auto lower = *it;
114 double t = (time - lower.first) / (upper.first - lower.first);
115 double scale1 = upper.second;
116 double scale0 = lower.second;
117 return scale0 + t * (scale1 - scale0);
118 }
119}

142template <int SPACE_DIM> MoFEMErrorCode TimeScaleVector<SPACE_DIM>::timeData() {
143 MoFEMFunctionBegin;
144
145 char time_file_name[255];
146 PetscBool flg = PETSC_FALSE;
147 if (!nAme.empty())
148 CHKERR PetscOptionsGetString(PETSC_NULL, PETSC_NULL, nAme.c_str(),
149 time_file_name, 255, &flg);
150
151 if (!flg) {
152 if (errorIfFileNotGiven)
153 SETERRQ1(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
154 "*** ERROR %s (time_data FILE NEEDED)", nAme.c_str());
155 MoFEMFunctionReturnHot(0);
156 }
157
158 FILE *time_data = fopen(time_file_name, "r");
159 if (time_data == NULL) {
160 SETERRQ1(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
161 "*** ERROR data file < %s > open unsuccessful", time_file_name);
162 }
163 double no1 = 0.0;
164 FTensor::Index<'i', 3> i;
165 (tSeries[no1])(i) = 0.;
166 while (!feof(time_data)) {
167 FTensor::Tensor1<double, 3> no2{0., 0., 0.};
168 int n =
169 fscanf(time_data, "%lf %lf %lf %lf", &no1, &no2(0), &no2(1), &no2(2));
170 if (n < 0) {
171 fgetc(time_data);
172 continue;
173 }
174 if (n != 4) {
175 SETERRQ1(PETSC_COMM_SELF, MOFEM_DATA_INCONSISTENCY,
176 "*** ERROR read data file error (check input time data file) "
177 "{ n = %d }",
178 n);
179 }
180 (tSeries[no1])(i) = no2(i);
181 }
182 int r = fclose(time_data);
183
184 MOFEM_LOG_CHANNEL("WORLD");
185 for (auto &[ts, vec] : tSeries) {
186 MOFEM_TAG_AND_LOG_C("WORLD", Sev::verbose, "TimeScaleVector",
187 "** read vector %3.2e time %3.2e %3.2e %3.2e", ts,
188 vec(0), vec(1), vec(2));
189 }
190 MOFEM_LOG_CHANNEL("WORLD");
191
192 if (r != 0) {
193 SETERRQ(PETSC_COMM_SELF, MOFEM_INVALID_DATA,
194 "*** ERROR file close unsuccessful");
195 }
196 readFile = 1;
197
198 if (readFile == 1)
199 scalingMethod = [this](double time) {
200 return this->getVectorFromData(time);
201 };
202
203 MoFEMFunctionReturn(0);
204}

212template <int SPACE_DIM>
213FTensor::Tensor1<double, SPACE_DIM>
214TimeScaleVector<SPACE_DIM>::getVectorFromData(const double time) {
215
216 FTensor::Tensor1<double, SPACE_DIM> Nf;
217 FTensor::Tensor1<double, 3> acc;
218 FTensor::Tensor1<double, 3> acc0 = tSeries.begin()->second;
219
220 FTensor::Index<'I', SPACE_DIM> I;
221 FTensor::Index<'i', SPACE_DIM> i;
222
223 double t0 = 0, t1, dt;
224 for (auto &[ts, vec] : tSeries) {
225 if (ts > time) {
226 t1 = ts;
227 dt = time - t0;
228 acc(I) = acc0(I) + ((vec(I) - acc0(I)) / (t1 - t0)) * dt;
229 break;
230 }
231 t0 = ts;
232 acc0 = vec;
233 acc = acc0;
234 }
235 Nf(i) = acc(i);
236 return Nf;
237}
