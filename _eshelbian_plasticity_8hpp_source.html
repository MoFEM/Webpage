362 OpAssembleBasic(
363 const std::string &row_field, const std::string &col_field,
364 boost::shared_ptr<DataAtIntegrationPts> data_ptr, const char type,
365 const bool assemble_symmetry, ScaleOff scale_off = []() { return 1; })
366 : T(row_field, col_field, type, false), dataAtPts(data_ptr),
367 assembleSymmetry(assemble_symmetry), scaleOff(scale_off) {}

369 VectorDouble nF; ///< local right hand side vector

389 MoFEMFunctionBegin;
390 double *vec_ptr = &*nF.begin();
391 int nb_dofs = data.getIndices().size();
392 int *ind_ptr = &*data.getIndices().begin();
393 CHKERR VecSetValues(this->getTSf(), nb_dofs, ind_ptr, vec_ptr, ADD_VALUES);
394 MoFEMFunctionReturn(0);
395 }

399 MoFEMFunctionBegin;
400 CHKERR assemble(data);
401 MoFEMFunctionReturn(0);
402 }
403
404 virtual MoFEMErrorCode integrate(EntData &row_data, EntData &col_data) {
405 MoFEMFunctionBegin;
406 SETERRQ(PETSC_COMM_SELF, MOFEM_NOT_IMPLEMENTED, "Not yet implemented");
407 MoFEMFunctionReturn(0);
408 }
409
410 virtual MoFEMErrorCode assemble(int row_side, int col_side,
411 EntityType row_type, EntityType col_type,
412 EntData &row_data, EntData &col_data) {
413 MoFEMFunctionBegin;
414
415 if (assembleSymmetry) {
416 const auto row_nb_dofs = row_data.getIndices().size();
417 const auto col_nb_dofs = col_data.getIndices().size();
418 transposeK.resize(col_nb_dofs, row_nb_dofs, false);
419 noalias(transposeK) = trans(K);
420 transposeK *= scaleOff();
421 }
422
423 CHKERR MatSetValues<SchurL2Mats>(this->getTSA(), row_data, col_data, K,
424 ADD_VALUES);
425 if (assembleSymmetry) {
426 CHKERR MatSetValues<SchurL2Mats>(this->getTSA(), col_data, row_data,
427 transposeK, ADD_VALUES);
428 }
429
430 MoFEMFunctionReturn(0);
431 }
432
433 MoFEMErrorCode doWork(int side, EntityType type, EntData &data) {
434 MoFEMFunctionBegin;
435 if (data.getIndices().empty())
436 MoFEMFunctionReturnHot(0);
437 nF.resize(data.getIndices().size(), false);
438 nF.clear();
439 CHKERR integrate(side, type, data);
440 CHKERR assemble(side, type, data);
441 MoFEMFunctionReturn(0);
442 }
443
444 MoFEMErrorCode doWork(int row_side, int col_side, EntityType row_type,
445 EntityType col_type, EntData &row_data,
446 EntData &col_data) {
447 MoFEMFunctionBegin;
448 if (row_data.getIndices().empty())
449 MoFEMFunctionReturnHot(0);
450 if (col_data.getIndices().empty())
451 MoFEMFunctionReturnHot(0);
452 K.resize(row_data.getIndices().size(), col_data.getIndices().size(), false);
453 K.clear();
454 CHKERR integrate(row_data, col_data);
455 CHKERR assemble(row_side, col_side, row_type, col_type, row_data, col_data);
456 MoFEMFunctionReturn(0);
457 }
